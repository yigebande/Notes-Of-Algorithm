# 39. 组合总和

思路是很简单的，如果含当前元素的符合要求的集合没有找完，那就继续使用当前元素作为起点进行遍历；找完后，以下一个元素为起点进行遍历。

我后面根据文章讲解的剪枝修改了一下，这个排序很重要
```c++
sort(candidates.begin(), candidates.end());
```


```c++
class Solution {
public:
    
    void backTracking(vector<int>& candidates, vector<int>& path, 
                      vector<vector<int>>& res, int target, int start) {
        if (target == 0) {
            res.push_back(path);
            return;
        }
        for (int i = start; i < candidates.size() && target - candidates[i] >= 0; i++) {
            path.push_back(candidates[i]);
            backTracking(candidates, path, res, target - candidates[i], i);
            path.pop_back();
        }
        return;
    }

    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        vector<vector<int>> res;
        vector<int>path;
        sort(candidates.begin(), candidates.end());
        backTracking(candidates, path, res, target, 0);
        return res;
    }
};
```
