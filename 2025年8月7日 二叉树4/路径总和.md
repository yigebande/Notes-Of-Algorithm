# 112. 路径总和

很简单的一道题。

dfs

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        if (root == nullptr) {
            return false;
        }
        if (targetSum - root->val == 0) {
            if (!root->left && !root->right) return true;
        }
        return hasPathSum(root->left, targetSum - root->val) || 
               hasPathSum(root->right, targetSum - root->val);
    }
};
```

# 113. 路径总和II

思路和上面类似。

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void helper(TreeNode* root, int targetSum, vector<int>& path, vector<vector<int>>& vec) {
        if (root == nullptr) return;    // 如果节点为空，那就不处理
        if (targetSum - root->val == 0) {   // 如果头节点到当前节点的总和等于targetSum
            if (!root->left && !root->right) {  // 如果当前节点是叶子节点，记录这条路径
                path.push_back(root->val);
                vec.push_back(path);
                path.pop_back();
            }
        }
        if (root->left) {
            path.push_back(root->val);
            helper(root->left, targetSum - root->val, path, vec);
            path.pop_back();
        }
        if (root->right) {
            path.push_back(root->val);
            helper(root->right, targetSum - root->val, path, vec);
            path.pop_back();
        }
    }

    vector<vector<int>> pathSum(TreeNode* root, int targetSum) {
        vector<vector<int>> res;
        vector<int> path;
        helper(root, targetSum, path, res);
        return res;
    }
};
```

每一次把新节点添加进`path`并且进行完`helper`之后，都要`path.pop_back()`一下，因为这是回溯。
