# 450. 删除二叉搜索树中的节点

看了文章讲解做的，最难想到的就是根节点的左右子树都存在的时候应该如何处理。

文章讲解给的方案是，把根节点的左树连接到根节点的右树的最左边的子节点上，如果不看答案真的想不出来。

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:

    TreeNode* deleteNode(TreeNode* root, int key) {
        // 0. 如果节点为空，说明没找到和key相等的root->val
        if (root == nullptr) return root;
        if (root->val == key) {
            // 1. 根节点的两个子节点为空
            if (root->left == nullptr && root->right == nullptr) {
                delete root;
                return nullptr;
                // 2. 左子节点不为空，右子节点为空，返回左子节点
            } else if (root->left != nullptr && root->right == nullptr) {
                TreeNode* left = root->left;
                delete root;
                return left;
                // 3. 右子节点不为空，左子节点为空，返回右子节点
            } else if (root->left == nullptr && root->right != nullptr) {
                TreeNode* right = root->right;
                delete root;
                return right;
                // 4. 左右子节点均不为空，把左树连接到右树的左树的末叶
            } else {
                TreeNode* leftmostLeafOfRight = root->right;
                TreeNode* left = root->left;
                TreeNode* right = root->right;
                while (leftmostLeafOfRight->left != nullptr) {
                    leftmostLeafOfRight = leftmostLeafOfRight->left;
                }
                delete root;
                leftmostLeafOfRight->left = left;
                return right;
            }
        }
        root->left = deleteNode(root->left, key);
        root->right = deleteNode(root->right, key);
        return root;
    }
};
```
