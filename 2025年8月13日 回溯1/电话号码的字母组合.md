# 17. 电话号码的字母组合

很简单的一道题

```c++
class Solution {
public:
    vector<string> numbers {
        "",
        "",
        "abc",
        "def",
        "ghi",
        "jkl",
        "mno",
        "pqrs",
        "tuv",
        "wxyz"
    };
    void backTracking(string digits, const string path, vector<string>& res) {
        if (digits == "") {
            // 这一句很重要，因为如果一开始digits为空，就不能把""加入到res中
            if (path != "") res.push_back(path);
            return;
        }
        // 取digit的第一个数字里的元素
        for (auto& ch: numbers[digits[0] - '0']) {
            backTracking(digits.substr(1), path + ch, res);
        }
    }

    vector<string> letterCombinations(string digits) {
        vector<string> res;
        backTracking(digits, "", res);
        return res;
    }
};
```
