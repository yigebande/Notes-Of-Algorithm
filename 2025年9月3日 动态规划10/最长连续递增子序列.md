# 674. 最长连续递增子序列
非常简单的一道题，只需要从头到尾一直遍历就行
```c++
class Solution {
public:
    int findLengthOfLCIS(vector<int>& nums) {
        int len = 1;    // 当前遍历的连续递增子序列的长度
        int maxLen = len;   // 历史记录中最长连续递增子序列的长度
        for (int i = 1; i < nums.size(); i++) {
            if (nums[i] > nums[i - 1]) {
                len++;
            } else {
                len = 1;
            }
            maxLen = max(maxLen, len);
        }
        return maxLen;
    }
};
```
