# 739. 每日温度
单调栈的入门题目
```c++
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        vector<int> arr(temperatures.size(), 0);
        stack<int> st;
        st.push(0);
        for (int i = 1; i < temperatures.size(); i++) {
            while (!st.empty() && temperatures[i] > temperatures[st.top()]) {
                arr[st.top()] = i - st.top();
                st.pop();
            }
            st.push(i);
        }
        return arr;
    }
};
```

从栈顶到栈底，元素大小由小变大。
