# 455. 分发饼干

一道简单的题目，我的思路是先用小饼干喂饱小胃袋
```c++
class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        if (s.size() == 0) return 0;
        sort(g.begin(), g.end());
        sort(s.begin(), s.end());
        int childIndex = 0;
        for (int i = 0; i < s.size(); i++) {
            if (childIndex == g.size()) break;
            if (s[i] >= g[childIndex]) {
                childIndex++;
            } else {
                continue;
            }
        }
        return childIndex;
    }
};
```
